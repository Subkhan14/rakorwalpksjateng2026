async function handleFormSubmit(formObject) { var btn = document.getElementById('btnSubmit'); var container = document.getElementById('formContainer'); btn.disabled = true; btn.innerText = "Memproses..."; // 1. Mengubah form ke format yang bisa dikirim ke Apps Script const formData = new URLSearchParams(); new FormData(formObject).forEach((value, key) => { formData.append(key, value); }); try { // 2. Kirim data ke Apps Script via POST // Ganti URL di bawah dengan URL /exec milik Anda const response = await fetch("https://script.google.com/macros/s/AKfycbynx4Sppg_xvj_-QlJwGBlFIIS9UKucG5Cih2as1upgrL1ErSbCspFaFRFjr5pyTdHQ/exec", { method: "POST", body: formData }); const result = await response.json(); if (result.status === "success") { window.scrollTo({ top: 0, behavior: 'smooth' }); container.innerHTML = `
✓

Terima kasih, data Anda telah terekam.

`; } else { throw new Error(result.pesan || "Gagal mengirim data"); } } catch (error) { var msgBox = document.getElementById('outputMessage'); msgBox.classList.remove('hidden'); msgBox.className = "m-5 p-4 rounded-md text-sm font-medium text-center bg-red-100 text-red-800"; msgBox.innerText = error.message; btn.disabled = false; btn.innerText = "Submit"; } }
